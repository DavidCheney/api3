# parameterized query on a specific VPDS
query DataSphere($vpdsUoid:String) {
  virtualDataSphere(uoid:$vpdsUoid) {
    id {
      ...ResourceId
    }
    location
    admins {
      ...UserAuth
    }
    health {
      ...Health
    }
    computeResources {
      cpu
    }
    communicationResources {
      networkInterfaces {
        ...NetworkInterface
      }
    }
    storageResources {
      ...StorageResources
    }
  }
}

query VPDS_Resources {
  dataSphere {
    virtualDataSpheres {
      id {
	      ...ResourceId
       }
      storageResources {
        ...StorageResources
      }
    }
  }
}

query VPDS_Mobility {
  dataSphere {
    virtualDataSpheres {
      id {
        ...ResourceId
      }
      storageResources {
        storageGroups {
          mobility {
            ...Mobility
            mostActiveChildUoids
            leastActiveChildUoids
          }
        }
      }
    }
  }
}
